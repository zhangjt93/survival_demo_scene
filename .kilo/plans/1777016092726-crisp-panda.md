# 修复游戏画面抖动问题

## 问题分析

经过代码审查，画面抖动（屏幕晃动）的主要原因：

1. **velocity 突变**：角色移动时直接 `velocity.x = direction.x * SPEED`，瞬间跳变到满速，没有平滑过渡
2. **Camera3D 刚性跟随**：相机是 CharacterBody3D 的子节点，角色的任何微小位置跳动（Jolt Physics 接触解算微调、物理步进与渲染帧不同步）都直接传递给相机
3. **没有位置平滑**：没有为相机添加插值平滑（position smoothing / lagging following）

## 修复方案

### 修改 `scripts/character_controller.gd`

#### 改动 1：添加加速度/减速平滑
- 引入 `ACCELERATION` 和 `FRICTION` 常量
- 移动时使用 `move_toward` 平滑加速而非直接赋值
- 停止时保持现有的 `move_toward` 减速（适当调整系数）

```gdscript
const SPEED = 5.0
const ACCELERATION = 15.0  # 新增：加速度
const FRICTION = 15.0      # 新增：摩擦/减速度

# 修改后的移动逻辑
if direction:
    velocity.x = move_toward(velocity.x, direction.x * SPEED, ACCELERATION * delta)
    velocity.z = move_toward(velocity.z, direction.z * SPEED, ACCELERATION * delta)
else:
    velocity.x = move_toward(velocity.x, 0, FRICTION * delta)
    velocity.z = move_toward(velocity.z, 0, FRICTION * delta)
```

#### 改动 2：为 Camera3D 添加位置平滑
- 用 `@onready var camera` 替代 `_ready()` 中手动获取
- 在 `_process(delta)` 中为相机高度位置添加 `lerp` 平滑插值

### 改动 3：清理旋转逻辑
- 移除 `orthonormalize()` 调用（对 CharacterBody3D 的 Basis 无用）
- 移除已注释的 `rotate_y(move.x)` 行

## 最终代码

```gdscript
extends CharacterBody3D
class_name CharacterController

const SPEED = 5.0
const JUMP_VELOCITY = 4.5
const ACCELERATION = 15.0
const FRICTION = 15.0
const CAMERA_SMOOTH_SPEED = 10.0
const CAMERA_Y_OFFSET = 1.232

@export var camera_mouse_rotation_speed:float = 0.001
@export var camera_x_rot_min:float = -89.9
@export var camera_x_rot_max:float = 70

@onready var camera:Camera3D = get_viewport().get_camera_3d()

func _ready() -> void:
	change_mouse_mode()

func _process(delta: float) -> void:
	camera.position.y = lerp(camera.position.y, CAMERA_Y_OFFSET, 1.0 - exp(-CAMERA_SMOOTH_SPEED * delta))

func _physics_process(delta: float) -> void:
	if not is_on_floor():
		velocity += get_gravity() * delta

	if Input.is_action_just_pressed("jump") and is_on_floor():
		velocity.y = JUMP_VELOCITY

	var input_dir := Input.get_vector("move_left", "move_right", "move_forward", "move_backward")
	var direction := (transform.basis * Vector3(input_dir.x, 0, input_dir.y)).normalized()
	if direction:
		velocity.x = move_toward(velocity.x, direction.x * SPEED, ACCELERATION * delta)
		velocity.z = move_toward(velocity.z, direction.z * SPEED, ACCELERATION * delta)
	else:
		velocity.x = move_toward(velocity.x, 0, FRICTION * delta)
		velocity.z = move_toward(velocity.z, 0, FRICTION * delta)

	move_and_slide()

func _unhandled_input(event: InputEvent) -> void:
	var scale_factor:float = min(
		(float(get_viewport().size.x) / get_viewport().get_visible_rect().size.x),
		(float(get_viewport().size.y) / get_viewport().get_visible_rect().size.y)
	)
	if event is InputEventMouseMotion:
		var camera_rotation = event.relative * camera_mouse_rotation_speed * scale_factor
		rotate_node(-camera_rotation)

func rotate_node(move:Vector2):
	rotate_y(move.x)
	orthonormalize()
	camera.rotation.x = clamp(camera.rotation.x + move.y, deg_to_rad(camera_x_rot_min), deg_to_rad(camera_x_rot_max))

func change_mouse_mode():
	if Input.mouse_mode == Input.MOUSE_MODE_CAPTURED:
		Input.set_mouse_mode(Input.MOUSE_MODE_VISIBLE)
	else:
		Input.set_mouse_mode(Input.MOUSE_MODE_CAPTURED)

func _input(event: InputEvent) -> void:
	if event.is_action_pressed("quit"):
		change_mouse_mode()
```

## 执行步骤

1. 编辑 `scripts/character_controller.gd`：
   - 添加 `ACCELERATION`、`FRICTION`、`CAMERA_SMOOTH_SPEED`、`CAMERA_Y_OFFSET` 常量
   - 将 `var camera` 改为 `@onready var camera`
   - 移除 `_ready()` 中的 `camera = get_viewport().get_camera_3d()` 行
   - 修改 `_physics_process` 中的移动逻辑为 `move_toward` 加速/减速
   - 添加 `_process(delta)` 实现相机 Y 轴位置平滑
   - 保留现有旋转逻辑（`orthonormalize` 等不变）
2. 运行游戏测试，验证抖动是否消除，手感是否正常

## 预期效果

- 角色移动和停止更平滑，不再有瞬间速度跳变
- 相机 Y 轴位置带有平滑跟随，过滤掉角色垂直方向的微小波动
- 整体画面观感更稳定，不晕
